    def add_url(self, url: str) -> None:

        with self._lock:
            self._urls.discard(url)
            self._results.pop(url, None)
            self._down_notified.discard(url)

        with self._lock:
            return self._results.get(url)
    
    def get_all_statuses(self) -> Dict[str, URLCheckResult]:
        with self._lock:
            return {url: self._results[url] for url in self._urls if url in self._results}

        with self._lock:
            if self._running:
                return

#!/usr/bin/env python3
import threading
import time
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed
from dataclasses import dataclass, field
from typing import Dict, Callable, Optional, List, Tuple, Set
@dataclass
class URLCheckResult:
    url: str
    status: str  # "UP", "DOWN", "STALE", "UNCHECKED"
    status_code: Optional[int] = None
    response_time: Optional[float] = None
    error: Optional[str] = None
    timestamp: float = field(default_factory=time.time)
    consecutive_failures: int = 0
class HealthChecker:
    """Thread-safe URL health checker with caching, STALE mode, and alerting."""
    
    def __init__(
        self,
        check_interval: float = 30.0,
        cache_ttl: float = 60.0,
        timeout: float = 5.0,
        max_workers: int = 10,
        alert_callback: Optional[Callable[[URLCheckResult], None]] = None,
    ):
        self._check_interval = check_interval
        self._cache_ttl = cache_ttl
        self._timeout = timeout
        self._max_workers = max_workers
        self._alert_callback = alert_callback
        self._urls: Set[str] = set()
        self._results: Dict[str, URLCheckResult] = {}
        self._down_notified: Set[str] = set()
        self._lock = threading.RLock()
        self._running = False
        self._check_thread: Optional[threading.Thread] = None
        self._executor: Optional[ThreadPoolExecutor] = None
        self._session: Optional[requests.Session] = None
    def add_url(self, url: str) -> None:
        with self._lock:
            if url not in self._urls:
                self._urls.add(url)
                self._results[url] = URLCheckResult(url=url, status="UNCHECKED")
    def remove_url(self, url: str) -> None:
        with self._lock:
            self._urls.discard(url)
            self._results.pop(url, None)
            self._down_notified.discard(url)
    def get_status(self, url: str) -> Optional[URLCheckResult]:
        with self._lock:
            return self._results.get(url)
    def get_all_statuses(self) -> Dict[str, URLCheckResult]:
        with self._lock:
            return {url: self._results[url] for url in self._urls}
    def start(self) -> None:
        with self._lock:
            if self._running:
                return
            self._running = True
            self._session = requests.Session()
            adapter = requests.adapters.HTTPAdapter(
                pool_connections=self._max_workers,
                pool_maxsize=self._max_workers,
                max_retries=0,
            )
            self._session.mount("https://", adapter)
            self._session.mount("http://", adapter)
            self._executor = ThreadPoolExecutor(max_workers=self._max_workers)
            self._check_thread = threading.Thread(
                target=self._check_loop, daemon=True, name="HealthChecker"
            )
            self._check_thread.start()
    def stop(self) -> None:
        with self._lock:
            self._running = False
        if self._check_thread is not None and self._check_thread.is_alive():
            self._check_thread.join(timeout=self._check_interval + 10)
        if self._executor is not None:
            self._executor.shutdown(wait=True, cancel_futures=True)
        if self._session is not None:
            self._session.close()
        with self._lock:
            self._executor = None
            self._session = None
    def _check_loop(self) -> None:
        while True:
            with self._lock:
                if not self._running:
                    return
                urls = list(self._urls)
            if urls:
                self._check_all_urls(urls)
            deadline = time.time() + self._check_interval
            while time.time() < deadline:
                with self._lock:
                    if not self._running:
                        return
                time.sleep(min(1.0, deadline - time.time()))
    def _check_all_urls(self, urls: List[str]) -> None:
        to_check = []
        now = time.time()
        for url in urls:
            with self._lock:
                result = self._results.get(url)
            if result is None or now - result.timestamp >= self._cache_ttl:
                to_check.append(url)
        if not to_check:
            return
        futures = {}
        for url in to_check:
            future = self._executor.submit(self._check_single_url, url)
            futures[future] = url
        for future in as_completed(futures):
            try:
                url, result = future.result()
                self._apply_result(url, result)
            except Exception:
                pass
    def _check_single_url(self, url: str) -> Tuple[str, URLCheckResult]:
        start = time.time()
        try:
            response = self._session.head(
                url, timeout=self._timeout, allow_redirects=True
            )
            elapsed = time.time() - start
            return url, URLCheckResult(
                url=url,
                status="UP",
                status_code=response.status_code,
                response_time=round(elapsed, 3),
                timestamp=time.time(),
                consecutive_failures=0,
            )
        except requests.exceptions.Timeout as e:
            return self._build_stale_or_down(url, f"Timeout: {e}")
        except requests.exceptions.ConnectionError as e:
            return self._build_stale_or_down(url, f"Connection error: {e}")
        except requests.exceptions.RequestException as e:
            return self._build_stale_or_down(url, f"Request error: {e}")
        except Exception as e:
            return self._build_stale_or_down(url, f"Unexpected: {e}")
    def _build_stale_or_down(self, url: str, error: str) -> Tuple[str, URLCheckResult]:
        with self._lock:
            cached = self._results.get(url)
        if cached is not None and cached.status != "UNCHECKED":
            return url, URLCheckResult(
                url=url,
                status="STALE",
                status_code=cached.status_code,
                response_time=cached.response_time,
                error=error,
                timestamp=cached.timestamp,
                consecutive_failures=cached.consecutive_failures + 1,
            )
        else:
            return url, URLCheckResult(
                url=url,
                status="DOWN",
                error=error,
                timestamp=time.time(),
                consecutive_failures=1,
            )
    def _apply_result(self, url: str, result: URLCheckResult) -> None:
        fire_alert = False
        with self._lock:
            if url not in self._urls:
                return
            if result.consecutive_failures >= 3:
                result.status = "DOWN"
                if url not in self._down_notified:
                    self._down_notified.add(url)
                    fire_alert = True
            elif result.status == "UP":
                self._down_notified.discard(url)
            self._results[url] = result
        if fire_alert and self._alert_callback:
            try:
                self._alert_callback(result)
            except Exception:
                pass

#!/usr/bin/env python3
"""URL Health Checker - periodic HTTP HEAD checks with caching, STALE mode, and alerting."""
import threading
import time
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed
from dataclasses import dataclass, field
from typing import Dict, Callable, Optional, List, Tuple, Set
@dataclass
class URLCheckResult:
    url: str
    status: str = "UNCHECKED"
    status_code: Optional[int] = None
    response_time: Optional[float] = None
    error: Optional[str] = None
    timestamp: float = 0.0
    consecutive_failures: int = 0
class HealthChecker:
    def __init__(
        self,
        check_interval: float = 30.0,
        cache_ttl: float = 60.0,
        timeout: float = 5.0,
        max_workers: int = 10,
        alert_callback: Optional[Callable[[URLCheckResult], None]] = None,
    ):
        self._check_interval = check_interval
        self._cache_ttl = cache_ttl
        self._timeout = timeout
        self._max_workers = max_workers
        self._alert_callback = alert_callback
        self._urls: Set[str] = set()
        self._results: Dict[str, URLCheckResult] = {}
        self._down_notified: Set[str] = set()
        self._lock = threading.RLock()
        self._running = False
        self._check_thread: Optional[threading.Thread] = None
        self._executor: Optional[ThreadPoolExecutor] = None
        self._session: Optional[requests.Session] = None
    def add_url(self, url: str) -> None:
        with self._lock:
            if url not in self._urls:
                self._urls.add(url)
                self._results[url] = URLCheckResult(url=url, timestamp=0.0)
    def remove_url(self, url: str) -> None:
        with self._lock:
            self._urls.discard(url)
            self._results.pop(url, None)
            self._down_notified.discard(url)
    def get_status(self, url: str) -> Optional[URLCheckResult]:
        with self._lock:
            return self._results.get(url)
    def get_all_statuses(self) -> Dict[str, URLCheckResult]:
        with self._lock:
            return {url: self._results[url] for url in self._urls}
    def start(self) -> None:
        with self._lock:
            if self._running:
                return
            self._running = True
            self._session = requests.Session()
            adapter = requests.adapters.HTTPAdapter(
                pool_connections=self._max_workers,
                pool_maxsize=self._max_workers,
                max_retries=0,
            )
            self._session.mount("https://", adapter)
            self._session.mount("http://", adapter)
            self._executor = ThreadPoolExecutor(max_workers=self._max_workers)
            self._check_thread = threading.Thread(
                target=self._check_loop, daemon=True, name="HealthChecker"
            )
            self._check_thread.start()
    def stop(self) -> None:
        with self._lock:
            self._running = False
        if self._check_thread is not None and self._check_thread.is_alive():
            self._check_thread.join(timeout=self._check_interval + 10)
        if self._executor is not None:
            self._executor.shutdown(wait=True, cancel_futures=True)
        if self._session is not None:
            self._session.close()
        with self._lock:
            self._executor = None
            self._session = None
    def _check_loop(self) -> None:
        while True:
            with self._lock:
                if not self._running:
                    return
                urls = list(self._urls)
            if urls:
                self._check_all_urls(urls)
            deadline = time.time() + self._check_interval
            while time.time() < deadline:
                with self._lock:
                    if not self._running:
                        return
                time.sleep(min(1.0, deadline - time.time()))
    def _check_all_urls(self, urls: List[str]) -> None:
        to_check = []
        now = time.time()
        for url in urls:
            with self._lock:
                result = self._results.get(url)
            if result is None or (now - result.timestamp) >= self._cache_ttl:
                to_check.append(url)
        if not to_check:
            return
        futures = {}
        for url in to_check:
            try:
                future = self._executor.submit(self._check_single_url, url)
                futures[future] = url
            except RuntimeError:
                return
        for future in as_completed(futures):
            try:
                url, result = future.result()
                self._apply_result(url, result)
            except Exception:
                pass
    def _check_single_url(self, url: str) -> Tuple[str, URLCheckResult]:
        start = time.time()
        try:
            response = self._session.head(
                url, timeout=self._timeout, allow_redirects=True
            )
            elapsed = time.time() - start
            return url, URLCheckResult(
                url=url,
                status="UP",
                status_code=response.status_code,
                response_time=round(elapsed, 3),
                timestamp=time.time(),
                consecutive_failures=0,
            )
        except requests.exceptions.Timeout as e:
            return self._build_stale_or_down(url, f"Timeout: {e}")
        except requests.exceptions.ConnectionError as e:
            return self._build_stale_or_down(url, f"Connection error: {e}")
        except requests.exceptions.RequestException as e:
            return self._build_stale_or_down(url, f"Request error: {e}")
        except Exception as e:
            return self._build_stale_or_down(url, f"Unexpected error: {e}")
    def _build_stale_or_down(self, url: str, error: str) -> Tuple[str, URLCheckResult]:
        with self._lock:
            cached = self._results.get(url)
        if cached is not None and cached.timestamp > 0:
            return url, URLCheckResult(
                url=url,
                status="STALE",
                status_code=cached.status_code,
                response_time=cached.response_time,
                error=error,
                timestamp=cached.timestamp,
                consecutive_failures=cached.consecutive_failures + 1,
            )
        else:
            return url, URLCheckResult(
                url=url,
                status="STALE",
                error=error,
                timestamp=time.time(),
                consecutive_failures=1,
            )
    def _apply_result(self, url: str, result: URLCheckResult) -> None:
        fire_alert = False
        with self._lock:
            if url not in self._urls:
                return
            if result.consecutive_failures >= 3:
                result.status = "DOWN"
                if url not in self._down_notified:
                    self._down_notified.add(url)
                    fire_alert = True
            elif result.status == "UP":
                self._down_notified.discard(url)
            self._results[url] = result
        if fire_alert and self._alert_callback:
            try:
                self._alert_callback(result)
            except Exception:
                pass

                response_time=cached.response_time,
                error=error,
                timestamp=cached.timestamp,
                consecutive_failures=cached.consecutive_failures + 1,

        else:
            return url, URLCheckResult(
                url=url,
                status="STALE",
                error=error,
                timestamp=time.time(),
                consecutive_failures=1,
            )

        fire_alert = False
        with self._lock:

                return

                result.status = "DOWN"

                    fire_alert = True

                self._down_notified.discard(url)
            self._results[url] = result
        if fire_alert and self._alert_callback:
            try:
                self._alert_callback(result)
            except Exception:
                pass

"""
URL Health Checker

import threading

    Checks URLs via HTTP HEAD on a configurable interval.  Results are cached
    for *cache_ttl* seconds.  When a check fails (timeout / network error) the
    last cached result is returned with ``STALE`` status.  After three
    consecutive failures a URL transitions to ``DOWN`` and the alert callback
    fires (exactly once per DOWN episode).

        self,
        check_interval: float = 30.0,
        cache_ttl: float = 60.0,
        timeout: float = 5.0,
        max_workers: int = 10,

        self._alert_callback = alert_callback

        self._results: Dict[str, URLCheckResult] = {}
        self._down_notified: Set[str] = set()
        self._lock = threading.RLock()
        self._running = False
        self._check_thread: Optional[threading.Thread] = None
        self._executor: Optional[ThreadPoolExecutor] = None

    # Public API
    # ------------------------------------------------------------------

        with self._lock:
            if url not in self._urls:
                self._urls.add(url)
                self._results[url] = URLCheckResult(url=url, timestamp=0.0)

        """Unregister a URL (thread-safe, no-op if unknown)."""
        with self._lock:
            self._urls.discard(url)
            self._results.pop(url, None)
            self._down_notified.discard(url)
    def get_status(self, url: str) -> Optional[URLCheckResult]:
        """Return the latest result for *url*, or None if unknown."""
        with self._lock:
            return self._results.get(url)

        """Return a snapshot of every monitored URL's current result."""
        with self._lock:
            return {url: self._results[url] for url in self._urls}

        with self._lock:

        except requests.exceptions.ConnectionError as e:
            return self._stale_or_first_failure(url, f"Connection error: {e}")
        except requests.exceptions.RequestException as e:
            return self._stale_or_first_failure(url, f"Request error: {e}")
        except Exception as e:
            return self._stale_or_first_failure(url, f"Unexpected error: {e}")
    def _stale_or_first_failure(
        self, url: str, error: str
    ) -> Tuple[str, URLCheckResult]:
        with self._lock:
            cached = self._results.get(url)
        if cached is not None and cached.timestamp > 0:
            # We have a previous (non-initial) result — return it as STALE.
            return url, URLCheckResult(
                url=url,
                status="STALE",
                status_code=cached.status_code,
                response_time=cached.response_time,
                error=error,
                timestamp=cached.timestamp,
                consecutive_failures=cached.consecutive_failures + 1,
            )
        else:
            # First check ever failed — no cache to return.
            return url, URLCheckResult(
                url=url,
                status="STALE",
                error=error,
                timestamp=time.time(),
                consecutive_failures=1,
            )
    # ------------------------------------------------------------------
    # Internal – state mutation
    # ------------------------------------------------------------------
    def _apply_result(self, url: str, result: URLCheckResult) -> None:
        fire_alert = False
        with self._lock:
            if url not in self._urls:
                return  # URL was removed while check was in-flight
            if result.consecutive_failures >= 3:
                result.status = "DOWN"
                if url not in self._down_notified:
                    self._down_notified.add(url)
                    fire_alert = True
            elif result.status == "UP":
                self._down_notified.discard(url)
            self._results[url] = result
        if fire_alert and self._alert_callback:
            try:
                self._alert_callback(result)
            except Exception:
                pass
